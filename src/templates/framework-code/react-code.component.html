import { useState } from 'react';

export default function Example() {
    const [name, setName] = useState('');

    return (
        <div>
            <label>Name:</label>
                <input
                    type="text"
                    value={name}
                    onChange={(e)=>setName(e.target.value)}
                    placeholder="Enter a name here"
                />
                <hr/>
            <h1>Hello {name}!</h1>
        </div>
    );
};
